// ========================================================================= // ENGINE AVANCÉ : DETECTION ET APPLICATION STRANGE DES COULEURS // ========================================================================= function startDragonAnalyse() { const timeValue = document.getElementById('dragon-time-input').value || "18:30"; let hours = parseInt(timeValue.split(':')[0]) || 18; let minutes = parseInt(timeValue.split(':')[1]) || 30; // Fenerateur de seed pseudo-aléatoire let seed = (hours * 60) + minutes; function randomSeeded() { let x = Math.sin(seed++) * 10000; return x - Math.floor(x); } // 1. Kajy ny salan'isa ankapobeny (Moyenne Dynamique) let avg = (6.0 + randomSeeded() * 6).toFixed(2); document.getElementById('avg-multiplier').innerText = `x${avg}`; let safe = (1.8 + randomSeeded() * 1.2).toFixed(2); let semi = (3.5 + randomSeeded() * 3.5).toFixed(2); let risque = (40.0 + randomSeeded() * 90).toFixed(2); document.getElementById('safe-val').innerText = `x${safe}`; document.getElementById('semi-val').innerText = `x${semi}`; document.getElementById('risque-val').innerText = `x${risque}`; // Ora ankapobeny eo amin'ny lohateny let currentSec = Math.floor(randomSeeded() * 45) + 10; document.getElementById('dynamic-prediction-time').innerText = `${String(hours).padStart(2, '0')}:${String(minutes).padStart(2, '0')}:${String(currentSec).padStart(2, '0')}`; function addSeconds(min, sec, addedSec) { let totalSec = min * 60 + sec + addedSec; return { min: Math.floor(totalSec / 60), sec: totalSec % 60 }; } function formatTimeStr(h, m, s) { let finalH = h + Math.floor(m / 60); return `${String(finalH % 24).padStart(2, '0')}:${String(m % 60).padStart(2, '0')}:${String(s).padStart(2, '0')}`; } // 2. Famoronana an'ireo Karatra Faminaniana 4 (Prediction Tours) let toursContainer = document.getElementById('prediction-tours-container'); toursContainer.innerHTML = ''; let currentTrackMin = minutes; let currentTrackSec = currentSec; for (let i = 1; i <= 4; i++) { let timeJump = Math.floor(75 + randomSeeded() * 65); let newTimeObj = addSeconds(currentTrackMin, currentTrackSec, timeJump); currentTrackMin = newTimeObj.min; currentTrackSec = newTimeObj.sec; let calculatedTimeStr = formatTimeStr(hours, currentTrackMin, currentTrackSec); let probability = Math.floor(45 + randomSeeded() * 45); // Famoronana ny multiplicateur an'ilay tour let randVal = randomSeeded(); let predMultiplier = 1.50; if (randVal > 0.85) { predMultiplier = (10.0 + randomSeeded() * 40).toFixed(2); // Rose } else if (randVal > 0.55) { predMultiplier = (2.0 + randomSeeded() * 7.9).toFixed(2); // Violet } else if (randVal > 0.35) { predMultiplier = (2.0 + randomSeeded() * 4.0).toFixed(2); // Vert } else { predMultiplier = (1.0 + randomSeeded() * 0.95).toFixed(2); // Bleu } // --- LOGIQUE DE COULEUR MAZAVA NY APPLICATION --- let borderClass = ""; let textTitleClass = ""; let probColorClass = ""; let bgGradient = ""; let shadowColor = ""; let badgeStyle = ""; if (parseFloat(predMultiplier) >= 10.0) { // 🌸 COULEUR ROSE borderClass = "border-2 border-pink-500"; textTitleClass = "text-pink-400 font-extrabold"; probColorClass = "text-pink-300"; bgGradient = "from-[#2d1222] to-[#0d1117]"; shadowColor = "rgba(236, 72, 153, 0.4)"; badgeStyle = "bg-pink-950 text-pink-300 border border-pink-700"; } else if (parseFloat(predMultiplier) >= 2.0) { if (randVal > 0.35 && randVal <= 0.55) { // 🟩 COULEUR VERT borderClass = "border-2 border-emerald-500"; textTitleClass = "text-emerald-400 font-extrabold"; probColorClass = "text-emerald-300"; bgGradient = "from-[#0a2016] to-[#0d1117]"; shadowColor = "rgba(16, 185, 129, 0.4)"; badgeStyle = "bg-emerald-950 text-emerald-300 border border-emerald-700"; } else { // 🟪 COULEUR VIOLET borderClass = "border-2 border-purple-500"; textTitleClass = "text-purple-400 font-extrabold"; probColorClass = "text-purple-300"; bgGradient = "from-[#1d122d] to-[#0d1117]"; shadowColor = "rgba(147, 51, 234, 0.4)"; badgeStyle = "bg-purple-950 text-purple-300 border border-purple-700"; } } else { // 🔷 COULEUR BLEU (Crash / ambany) borderClass = "border-2 border-blue-500"; textTitleClass = "text-blue-400 font-extrabold"; probColorClass = "text-blue-300"; bgGradient = "from-[#121b2d] to-[#0d1117]"; shadowColor = "rgba(59, 130, 246, 0.4)"; badgeStyle = "bg-blue-950 text-blue-300 border border-blue-700"; } // Injection an'ilay HTML node miaraka amin'ny loko vaovao let card = document.createElement('div'); card.className = `bg-gradient-to-b ${bgGradient} ${borderClass} rounded-2xl p-4 text-center space-y-2 transition duration-300 transform hover:scale-105`; card.style.boxShadow = `0 6px 20px ${shadowColor}`; card.innerHTML = `
TOUR ${i} • ${probability}%

${calculatedTimeStr}

CIBLE: ${predMultiplier}x
`; toursContainer.appendChild(card); } // 3. Famenoana ny tabilao OCR any aoriana const gridContainer = document.getElementById('ocr-grid-container'); gridContainer.innerHTML = ''; let roseCount = 0; let vertCount = 0; for (let i = 0; i < 24; i++) { let currentRandom = randomSeeded(); let baseVal = 1.01; if (currentRandom > 0.88) { baseVal = (10.0 + randomSeeded() * 60).toFixed(2); } else if (currentRandom > 0.60) { baseVal = (2.0 + randomSeeded() * 7).toFixed(2); } else { baseVal = (1.0 + randomSeeded() * 0.98).toFixed(2); } let colorClass = "text-cyan-400"; let borderClass = "bg-[#181f2e]"; let prefix = ""; if (parseFloat(baseVal) >= 10.0) { colorClass = "text-amber-400"; borderClass = "border border-yellow-500/40 bg-[#181f2e]"; prefix = "✨ "; roseCount++; } else if (parseFloat(baseVal) >= 2.0) { colorClass = "text-purple-400"; if (randomSeeded() > 0.75) { colorClass = "text-pink-500"; vertCount++; } } let span = document.createElement('span'); span.className = `${borderClass} ${colorClass} py-1.5 rounded text-[10px] font-mono font-bold`; span.innerText = `${prefix}${baseVal}x`; gridContainer.appendChild(span); } // Popup famantarana ora let popup = document.createElement('div'); popup.className = "absolute top-[40%] left-[15%] bg-gray-900 border border-gray-700 rounded px-2 py-1 text-[9px] flex items-center gap-1 shadow-2xl text-white font-mono"; popup.innerHTML = `🕒 ${formatTimeStr(hours, minutes, currentSec)}`; gridContainer.appendChild(popup); // Fanavaozana ny antontan'isa eo ambony document.getElementById('rose-count-stat').innerText = `${roseCount} (${(roseCount / 24 * 100).toFixed(1)}%)`; document.getElementById('vert-count-stat').innerText = `${vertCount} (${(vertCount / 24 * 100).toFixed(1)}%)`; document.getElementById('similaires-stat').innerText = `🏆 Similaires Détectés: ${Math.floor(1 + randomSeeded() * 4)}`; // Fampidinana ny pejy ho azy ho hita ny valiny const resultBox = document.getElementById('dragon-result-box'); resultBox.classList.remove('hidden'); resultBox.scrollIntoView({ behavior: 'smooth' }); }
